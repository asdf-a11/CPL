	CREATE_FUNCTION _printn, void, i32~_number
		OPENSCOPE 
		CREATE_ARGUMENT i32, _number
		CREATE i32, _bufferSize
		CREATE UNKNOWN, EXP_TEMP_VAR_0
		MOV EXP_TEMP_VAR_0, 11
		MOV _bufferSize, EXP_TEMP_VAR_0
		CREATE UNKNOWN, EXP_TEMP_VAR_1
		MOV EXP_TEMP_VAR_1, _bufferSize
		CREATELIST i32, _buffer, EXP_TEMP_VAR_1
		CREATE UNKNOWN, EXP_TEMP_VAR_2
		MOV EXP_TEMP_VAR_2, 0
		MOV _buffer, EXP_TEMP_VAR_2
		CREATE i32, _pointer
		CREATE UNKNOWN, EXP_TEMP_VAR_3
		MOV EXP_TEMP_VAR_3, 10
		MOV _pointer, EXP_TEMP_VAR_3
		CREATE i32, _runOnceFlag
		CREATE UNKNOWN, EXP_TEMP_VAR_4
		MOV EXP_TEMP_VAR_4, 1
		MOV _runOnceFlag, EXP_TEMP_VAR_4
		REPEAT rep_0
			OPENSCOPE 
			CREATE UNKNOWN, EXP_VAR_0
			CREATE i32, EXP_VAR_1
			!= EXP_VAR_1, _number, 0
			CREATE i32, EXP_VAR_2
			== EXP_VAR_2, _runOnceFlag, 1
			or EXP_VAR_0, EXP_VAR_1, EXP_VAR_2
			! EXP_VAR_0, EXP_VAR_0
			IF EXP_VAR_0
				OPENSCOPE 
				BREAK rep_0
				CLOSESCOPE 
			ENDIF 
			CREATE i32, _digit
			CREATE UNKNOWN, EXP_TEMP_VAR_5
			% EXP_TEMP_VAR_5, _number, 10
			MOV _digit, EXP_TEMP_VAR_5
			CREATE UNKNOWN, EXP_TEMP_VAR_6
			/ EXP_TEMP_VAR_6, _number, 10
			MOV _number, EXP_TEMP_VAR_6
			CREATE UNKNOWN, _#EXP_FUNCTION_ARGUMENT_4
			+ _#EXP_FUNCTION_ARGUMENT_4, _digit, 48
			CREATE UNKNOWN, _#EXP_FUNCTION_RETURN_5
			CALL _#EXP_FUNCTION_RETURN_5, _printc, _#EXP_FUNCTION_ARGUMENT_4
			CREATE UNKNOWN, _#EXP_FUNCTION_RETURN_6
			CALL _#EXP_FUNCTION_RETURN_6, _println
			CREATE UNKNOWN, EXP_TEMP_VAR_7
			MOV EXP_TEMP_VAR_7, _pointer
			CREATE UNKNOWN, EXP_TEMP_VAR_8
			+ EXP_TEMP_VAR_8, _digit, 48
			MOV _buffer, EXP_TEMP_VAR_8
			CREATE UNKNOWN, EXP_TEMP_VAR_9
			- EXP_TEMP_VAR_9, _pointer, 1
			MOV _pointer, EXP_TEMP_VAR_9
			CREATE UNKNOWN, EXP_TEMP_VAR_10
			MOV EXP_TEMP_VAR_10, 0
			MOV _runOnceFlag, EXP_TEMP_VAR_10
			CLOSESCOPE 
		ENDREPEAT 
		CREATE UNKNOWN, EXP_TEMP_VAR_11
		+ EXP_TEMP_VAR_11, _pointer, 1
		MOV _pointer, EXP_TEMP_VAR_11
		REPEAT rep_1
			OPENSCOPE 
			CREATE UNKNOWN, EXP_VAR_3
			< EXP_VAR_3, _pointer, _bufferSize
			! EXP_VAR_3, EXP_VAR_3
			IF EXP_VAR_3
				OPENSCOPE 
				BREAK rep_1
				CLOSESCOPE 
			ENDIF 
			CREATE UNKNOWN, _#EXP_FUNCTION_ARGUMENT_7
			CREATE UNKNOWN, _#EXP_INDEX_7
			MOV _#EXP_INDEX_7, _pointer
			* _#EXP_INDEX_7, _#EXP_INDEX_7, VAR_TYPE
			LDR _#EXP_INDEX_7, _buffer, _#EXP_INDEX_7
			MOV _#EXP_FUNCTION_ARGUMENT_7, _#EXP_INDEX_7
			CREATE UNKNOWN, _#EXP_FUNCTION_RETURN_9
			CALL _#EXP_FUNCTION_RETURN_9, _printc, _#EXP_FUNCTION_ARGUMENT_7
			CREATE UNKNOWN, EXP_TEMP_VAR_12
			+ EXP_TEMP_VAR_12, _pointer, 1
			MOV _pointer, EXP_TEMP_VAR_12
			CLOSESCOPE 
		ENDREPEAT 
		CLOSESCOPE 
	END_FUNCTION
